        ['var(--font-lineHeight-default)', 'FONT_LINE_HEIGHT_DEFAULT', "1.5"],
        ['var(--font-lineHeight-smallText)', 'FONT_LINE_HEIGHT_SMALL_TEXT', "1.8"],
        ['var(--font-size-xs)', 'FONT_SIZE_XS', "10px"],
        ['var(--font-size-s)', 'FONT_SIZE_S', "12px"],
        ['var(--font-size-m)', 'FONT_SIZE_M', "14px"],
        ['var(--font-size-default)', 'FONT_SIZE_DEFAULT', "14px"],
        ['var(--font-size-l)', 'FONT_SIZE_L', "16px"],
        ['var(--font-size-xl)', 'FONT_SIZE_XL', "20px"],
        ['var(--font-size-2xl)', 'FONT_SIZE_2XL', "24px"],
        ['var(--font-size-3xl)', 'FONT_SIZE_3XL', "32px"],
        ['var(--font-size-4xl)', 'FONT_SIZE_4XL', "38px"],
        ['var(--font-weight-default)', 'FONT_WEIGHT_DEFAULT', "400"],
        ['var(--font-weight-medium)', 'FONT_WEIGHT_MEDIUM', "500"],
        ['var(--font-weight-bold)', 'FONT_WEIGHT_BOLD', "700"],
    ].map(row => {
      const generateFontValueCell = () => {
        if (row[2].startsWith('rgba')){
          return {
            style: { padding: 0, color: mostReadableConfig(row[2]), height: "1px" }, 
            copyValue: row[2], 
            className: "transparentpattern", 
            children: <div style={{ height: "100%", padding: "16px", cursor: "pointer", backgroundColor: row[2] }}>{row[2]}</div> 
          }
        } else if (row[2].startsWith('#')){
          return {
            style: { padding: 0, color: mostReadableConfig(row[2]), height: "1px" }, 
            copyValue: row[2], 
            children: <div style={{ height: "100%", padding: "16px", cursor: "pointer", backgroundColor: row[2] }}>{row[2]}</div> 
          }
        } else if (row[1].startsWith('FONT_FAMILY')){
          return {
            children: <div style={{ fontFamily: row[2] }}>{row[2]}</div> 
          }
        } else if (row[1].startsWith('FONT_SIZE')){
          return {
            children: <div style={{ fontSize: row[2] }}>{row[2]}</div> 
          }
        } else if (row[1].startsWith('FONT_WEIGHT')){
          return {
            children: <div style={{ fontWeight: row[2] }}>{row[2]}</div> 
          }
        } else {
          return row[2]
        }
      }
      return [
        `\`${row[0]}\``,
        `\`${row[1]}\``,
        generateFontValueCell(),
      ]
    })} 
  />
